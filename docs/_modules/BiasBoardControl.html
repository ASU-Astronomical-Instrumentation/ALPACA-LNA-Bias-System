[docs]def MSBF(val: int) -> int:
    """Swaps byte order of 'val'. (needed for current sense chip)

    :param val: 16-bit value
    :type val: int (unsigned)
    :return: 16-bit MSB first value
    :rtype: int (unsigned)
    """
    return ((val & 0xFF) << 8) | ((val >> 8) & 0xFF)
